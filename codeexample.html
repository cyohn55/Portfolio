            # First the beginning lines of the scene are printed to the screen.
        
            print('\n Recite your baseline...\n
                   \n A blood black nothingness began to spin.

            # The two lines above correspond because they are both the last elements in 'q' and 'a'
        
            for baseline_qa in zip(q, a):       # Demonstrates the zip function
                print(baseline_qa)
        
        # Takes the variables 'questions' and 'answers' as arguments to the function and prints tuple pairs to the screen.
        blade_runner(questions, answers)
