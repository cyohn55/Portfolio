# Lines from the 'Voight Kampff test' in the sci-fi movie 'Blade Runner 2049' are initialized. 
# The scene involved a person asking the Blade Runner questions, followed by the Blade Runner's responses. 
# The 'Voight Kampff test' was meant to provoke an emotional response to determine whether the
# Blade Runner's artificial intelligence was becoming too intelligent and therefore dangerous.

# The variables 'questions' and 'answers' are initialized and set equal to the correct dialogue.
questions = ('Cells.',
             'Have you ever been in an institution? Cells.',
             'Do they keep you in a cell? Cells.',
             'When you're not performing your duties do they keep you in a little box? Cells.',
             'Interlinked.',
             'What's it like to hold the hand of someone you love? Interlinked.',
             'Do they teach you how to feel finger to finger? Interlinked.',
             'Do you long for having your heart interlinked? Interlinked.',
             'Do you dream about being interlinked?',
             'What's it like to hold your child in your arms? Interlinked.',
             'Do you feel that there's a part of you that's missing? Interlinked.',
             'Within cells interlinked.',
             'Why don't you say that three times? Within cells interlinked.',
             'We're done... Constant K. You can pick up your bonus.')

answers = ('Cells.', 'Cells.', 'Cells.', 'Cells',
           'Interlinked.', 'Interlinked.', 'Interlinked.', 'Interlinked.',
           'Interlinked.', 'Interlinked.', 'Interlinked.', 'Within cells interlinked.',
           'Within cells interlinked. ' * 3, 'Thank you, sir.')

# The function 'blade_runner' is defined. It takes two parameters 'q' and 'a'.

def blade_runner(q, a):
    # First the beginning lines of the scene are printed to the screen.
    print('\nRecite your baseline...\n'
          '\nA blood black nothingness began to spin.'
          '\nA system of cells interlinked.'
          '\nWithin cells interlinked.'
          '\nWithin cells interlinked.'
          '\nWithin one stem.'
          '\nAnd dreadfully distinct against the dark.'
          '\nA tall white fountain played.\n')

    # Iterates through each element in both 'q' and 'a' and uses the zip function to match each of the corresponding elements.
    for baseline_qa in zip(q, a):
        print(baseline_qa)

# Takes the variables 'questions' and 'answers' as arguments to the function and prints tuple pairs to the screen.
blade_runner(questions, answers)
